ls - для отображения всех файлов и папок
cd - переход на указанную директорию
mkdir - создание директории
touch - создание файла
nano filename - редактирование содержимового файла
cat - отображение содержимого файлов
grep - поиск по строкам в файлах
chmod +x script.sh - сделать скрипт исполняемым 
script.sh > output.log - запуск скрипта и перенаправление на output.log
